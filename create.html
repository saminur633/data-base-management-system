CREATE TABLE Product ( productID INTEGER PRIMARY KEY, category VARCHAR(50), price FLOAT ); CREATE TABLE Customer ( customerID INTEGER PRIMARY KEY, birthDate DATE, moneySpent INTEGER, anniversary DATE ); CREATE TABLE Orders ( orderID INTEGER PRIMARY KEY, customerID INTEGER, employeeID INTEGER, productID INTEGER, orderTotal FLOAT, orderDate DATE, FOREIGN KEY (customerID) REFERENCES Customer(customerID), FOREIGN KEY (productID) REFERENCES Product(productID) );
